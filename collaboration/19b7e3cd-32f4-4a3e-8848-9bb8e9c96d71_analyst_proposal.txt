import requests

def retrieve_collaboration_files(session_id, user_token):
    # Define the API endpoint for the collaboration platform
    api_endpoint = f"https://api.collaborationplatform.com/sessions/{session_id}/files"
    
    # Set up headers for authentication
    headers = {
        'Authorization': f'Bearer {user_token}',
        'Content-Type': 'application/json'
    }
    
    # Make the API request to retrieve files
    response = requests.get(api_endpoint, headers=headers)
    
    if response.status_code == 200:
        files = response.json()
        return files  # Return the list of files
    else:
        raise Exception(f"Error retrieving files: {response.status_code} - {response.text}")
